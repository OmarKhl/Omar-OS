TRADE LOG · SHEETS WRITE-BACK

When you log a trade here, it will also be appended to your Trade Log sheet automatically. Requires a one-time Apps Script setup.

SETUP INSTRUCTIONS ▾
1. Open your Google Sheet → Extensions → Apps Script
2. Paste the code below into a new script file
3. Click Deploy → New deployment → Web app
4. Set Execute as: Me · Who has access: Anyone
5. Copy the web app URL and paste it above 
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet()
                  .getSheetByName('Trade Log');
    if (!sheet) throw new Error('Sheet "Trade Log" not found');
    var last = sheet.getLastRow();
    sheet.appendRow([
      last - 1,            // Row #
      data.date,           // Date (MM-DD-YYYY)
      data.type,           // Transaction Type
      data.ticker,         // Ticker
      data.qty   || '',    // Quantity
      data.price || '',    // Amount per unit
      data.total || '',    // Total Amount
      data.fees  || 0,     // Trading Fees
      data.account || '',  // Investment Account
      data.split || '',    // Split Ratio
      data.fx    || 1,     // FX Rate
    ]);
    return ContentService
      .createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false,error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
